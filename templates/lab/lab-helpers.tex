% Helpers for short laboratory reports.
% In vkr.cls \section starts a new page. For laboratory reports this is allowed:
% short reports may stay compact, while larger reports can split chapters by page.

\renewcommand{\thefigure}{\arabic{figure}}
\renewcommand{\thetable}{\arabic{table}}

\newcounter{labSavedFigure}
\newcounter{labSavedTable}

\newcommand{\labsection}[1]{%
  \setcounter{labSavedFigure}{\value{figure}}%
  \setcounter{labSavedTable}{\value{table}}%
  \section*{#1}%
  \setcounter{figure}{\value{labSavedFigure}}%
  \setcounter{table}{\value{labSavedTable}}%
}

\newcommand{\labheading}[1]{%
  \par\noindent\textbf{#1}\par
}

\newcommand{\labinlineheading}[1]{%
  \par\noindent\textbf{#1} %
}

\newcommand{\labgoal}[1]{%
  \labsection{Цель работы}%
  \noindent #1\par
}

\newcommand{\labinitialdata}{%
  \labsection{Исходные данные (вариант \Вариант)}
}

\newcommand{\labwork}{%
  \labsection{Ход работы}
}

\newcommand{\labconclusion}[1]{%
  \labsection{Вывод}%
  \noindent #1\par
}

\newcommand{\labimage}[3][0.92\textwidth]{%
  \begin{figure}[H]
    \centering
    \includegraphics[width=#1]{#2}
    \caption{#3}
  \end{figure}
}
